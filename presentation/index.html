Dockerfile


FROM alpine:3.3

RUN apk add -U \
	bash \
	nginx \
	&& rm -rf /var/cache/apk*

# forward request and error logs to docker log collector
RUN ln -sf /dev/stdout /var/log/nginx/access.log
RUN ln -sf /dev/stderr /var/log/nginx/error.log

ADD /rootfs /
ADD /presentation /usr/share/nginx/html

CMD ["/bin/boot"]
EXPOSE 80
